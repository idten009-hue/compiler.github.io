Python Object-Oriented Programming (OOP) Tutorial

Last updated: June 18, 2026 | Reading time: 12 minutes

What is Object-Oriented Programming?

Object-Oriented Programming (OOP) is a programming paradigm that organizes code around objects and classes. It helps create modular, reusable, and maintainable code.

1. Classes and Objects

A class is a blueprint for creating objects. An object is an instance of a class.

class Dog:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def bark(self):
        print(f"{self.name} says: Woof!")

# Create objects
dog1 = Dog("Buddy", 3)
dog2 = Dog("Max", 5)

dog1.bark()      # Buddy says: Woof!
print(dog1.name) # Buddy

2. Attributes and Methods

Attributes are variables that belong to an object. Methods are functions that belong to a class.

class Car:
    def __init__(self, brand, model, year):
        # Attributes
        self.brand = brand
        self.model = model
        self.year = year
        self.speed = 0
    
    def accelerate(self, amount):
        self.speed += amount
        print(f"Speed: {self.speed} km/h")
    
    def brake(self):
        self.speed = 0
        print("Car stopped")
    
    def info(self):
        return f"{self.year} {self.brand} {self.model}"

car = Car("Toyota", "Camry", 2022)
print(car.info())          # 2022 Toyota Camry
car.accelerate(50)         # Speed: 50 km/h
car.accelerate(30)         # Speed: 80 km/h
car.brake()                # Car stopped

3. The __init__ Method (Constructor)

The __init__ method is called when an object is created. It initializes object attributes.

class Person:
    def __init__(self, name, age, email):
        self.name = name
        self.age = age
        self.email = email
        print(f"Person {name} created")

person = Person("Alice", 30, "alice@example.com")
# Output: Person Alice created

4. Encapsulation

Encapsulation is bundling data and methods. Python uses naming conventions for privacy.

class BankAccount:
    def __init__(self, balance):
        self.__balance = balance  # Private attribute
    
    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount
            print(f"Deposited: ${amount}")
    
    def withdraw(self, amount):
        if 0 < amount <= self.__balance:
            self.__balance -= amount
            print(f"Withdrawn: ${amount}")
        else:
            print("Insufficient funds")
    
    def get_balance(self):
        return self.__balance

account = BankAccount(1000)
account.deposit(500)           # Deposited: $500
print(account.get_balance())   # 1500
account.withdraw(200)          # Withdrawn: $200

5. Inheritance

Inheritance allows a class to inherit attributes and methods from another class.

class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        print(f"{self.name} makes a sound")

class Dog(Animal):
    def speak(self):
        print(f"{self.name} barks: Woof!")

class Cat(Animal):
    def speak(self):
        print(f"{self.name} meows: Meow!")

dog = Dog("Buddy")
cat = Cat("Whiskers")

dog.speak()    # Buddy barks: Woof!
cat.speak()    # Whiskers meows: Meow!

6. Method Overriding

Subclasses can override parent class methods.

class Vehicle:
    def start(self):
        print("Vehicle started")

class Car(Vehicle):
    def start(self):
        print("Car engine started with key")

class ElectricCar(Vehicle):
    def start(self):
        print("Electric car powered on")

vehicles = [Car(), ElectricCar()]
for vehicle in vehicles:
    vehicle.start()
# Output:
# Car engine started with key
# Electric car powered on

7. super() Function

Use super() to call methods from the parent class.

class Student:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def info(self):
        return f"Name: {self.name}, Age: {self.age}"

class CollegeStudent(Student):
    def __init__(self, name, age, student_id):
        super().__init__(name, age)
        self.student_id = student_id
    
    def info(self):
        parent_info = super().info()
        return f"{parent_info}, ID: {self.student_id}"

student = CollegeStudent("Bob", 20, "123456")
print(student.info())
# Output: Name: Bob, Age: 20, ID: 123456

8. Polymorphism

Polymorphism allows objects to take multiple forms.

class Shape:
    def area(self):
        pass

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        return 3.14 * self.radius ** 2

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height

shapes = [Circle(5), Rectangle(4, 6)]
for shape in shapes:
    print(f"Area: {shape.area()}")
# Output:
# Area: 78.5
# Area: 24

9. Class and Static Methods

class MathUtils:
    @staticmethod
    def add(a, b):
        return a + b
    
    @classmethod
    def from_string(cls, value_str):
        return cls(int(value_str))

# Use static method without creating instance
result = MathUtils.add(5, 3)
print(result)  # 8

10. Practical Example: Library System

class Book:
    def __init__(self, title, author, isbn):
        self.title = title
        self.author = author
        self.isbn = isbn
        self.available = True
    
    def __str__(self):
        return f"{self.title} by {self.author}"

class Library:
    def __init__(self):
        self.books = []
    
    def add_book(self, book):
        self.books.append(book)
    
    def borrow_book(self, isbn):
        for book in self.books:
            if book.isbn == isbn and book.available:
                book.available = False
                return f"You borrowed: {book}"
        return "Book not available"
    
    def return_book(self, isbn):
        for book in self.books:
            if book.isbn == isbn:
                book.available = True
                return f"You returned: {book}"

# Usage
library = Library()
book1 = Book("Python 101", "John Doe", "001")
book2 = Book("Data Science", "Jane Smith", "002")

library.add_book(book1)
library.add_book(book2)

print(library.borrow_book("001"))  # You borrowed: Python 101 by John Doe
print(library.return_book("001"))  # You returned: Python 101 by John Doe

Key Concepts Summary

  • Class: Blueprint for creating objects
  • Object: Instance of a class
  • Attribute: Variable belonging to an object
  • Method: Function belonging to a class
  • Encapsulation: Bundling data and methods
  • Inheritance: Creating classes from other classes
  • Polymorphism: Objects taking multiple forms
  • Abstraction: Hiding complex implementation details
